# `15.` 三数之和为 0

固定第一个，双指针移动后两个。

逻辑最清晰的写法：对 i、l、r 都先去重，保证它们都是一串相等的值的<font color="green">「第一个」</font>，再判断。循环体末尾，`l++` 或 `r--` 后，当时不用去重，等下次进入循环体后去重。为什么是「第一个」，以及去重方法的解释，参见 18 的题解。

```cpp
// leet ac 2023.10.19
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> v;
        int n = nums.size();
        if (n < 3) {
            return v;
        }
        sort(nums.begin(), nums.end());
        for (int i = 0; i <= n - 3; i++) {
            if (nums[i] > 0) {
                return v;
            }
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            } // i 去重结果：i 与 i-1 不同。i 是一串相同的里的第一个
            int l = i + 1, r = n - 1;
            while (l < r) {
                while (l > i + 1 && l < r && nums[l] == nums[l - 1]) {
                    l++;
                } // l 去重：一串相同的里的第一个。while 中的条件有 l>i+1，对 l 的初值 i+1 不成立，保证 l 可以取到 i+1
                if (l >= r) {
                    break;
                }
                while (r < n - 1 && r > l && nums[r] == nums[r + 1]) {
                    r--;
                } // r 去重：一串相同的里的第一个。while 中的条件有 r<n-1，对 r 的初值 n-1 不成立，保证 r 可以取到 n-1
                if (l >= r) {
                    break;
                }
                int sum = nums[i] + nums[l] + nums[r];
                if (sum == 0) {
                    v.push_back({nums[i], nums[l], nums[r]});
                    l++, r--; // 这里不用显式去重。在 while 的下一次循环中，会先去重。 
                } else {
                    sum < 0 ? l++ : r--; // 同上
                }
            }
        }
        return v;
    }
```

# `16.` 三数之和最接近某值

和不是 0，而是最接近某给定值。区别：`sum == target` 则直接返回，无需移动 l、r 了。（对比 15，`sum == 0` 时不返回，还需同时移动 l、r）

逻辑最清晰的写法：对 i、l、r 都先去重，保证它们都是一串相等的值的<font color="green">「第一个」</font>，再判断。后面移动 l、r 后，就不用去重了。

```cpp
// leet ac 2024.02.18
    int threeSumClosest(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int n = nums.size();

        int min_diff = INT_MAX;
        int ans;

        for (int i = 0; i <= n - 3; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            int l = i + 1, r = n - 1;
            while (l < r) {
                while (l > i + 1 && l < r && nums[l] == nums[l - 1]) {
                    l++;
                }
                if (l >= r) {
                    break;
                }
                while (r < n - 1 && l < r && nums[r] == nums[r + 1]) {
                    r--;
                }
                if (l >= r) {
                    break;
                }
                int sum = nums[i] + nums[l] + nums[r];
                if (sum == target) {
                    return target;
                }
                int t = abs(sum - target);
                if (t < min_diff) {
                    min_diff = t;
                    ans = sum;
                }
                sum < target ? l++ : r--;
            }
        }
        return ans;
    }
```
